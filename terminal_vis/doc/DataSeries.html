def add_data_set(data_set)
    @series << data_set

    if (@min_value == nil || @min_value > data_set.min_value)
        @min_value = data_set.min_value
    end
    if (@max_value == nil || @max_value < data_set.max_value)
        @max_value = data_set.max_value
    end
end
